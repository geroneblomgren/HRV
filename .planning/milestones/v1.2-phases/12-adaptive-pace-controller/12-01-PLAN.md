**_schedulerTick changes (line 99):**
- Remove the `halfPeriod` parameter
- Add `const halfPeriod = 1 / (AppState.pacingFreq * 2);` as the first line inside the function body
- Change the recursive setTimeout from `() => _schedulerTick(halfPeriod)` to just `_schedulerTick` (no argument)

Before:
```javascript
function _schedulerTick(halfPeriod) {
  while (_nextCueTime < _ctx.currentTime + SCHEDULE_AHEAD_SEC) {
    _scheduleCue(_nextCueTime, _nextPhase, halfPeriod);
    AppState.nextCueTime = _nextCueTime;
    AppState.nextCuePhase = _nextPhase;
    _nextCueTime += halfPeriod;
    _nextPhase = _nextPhase === 'inhale' ? 'exhale' : 'inhale';
  }
  _schedulerTimer = setTimeout(() => _schedulerTick(halfPeriod), LOOKAHEAD_MS);
}
```

After:
```javascript
function _schedulerTick() {
  const halfPeriod = 1 / (AppState.pacingFreq * 2);
  while (_nextCueTime < _ctx.currentTime + SCHEDULE_AHEAD_SEC) {
    _scheduleCue(_nextCueTime, _nextPhase, halfPeriod);
    AppState.nextCueTime = _nextCueTime;
    AppState.nextCuePhase = _nextPhase;
    _nextCueTime += halfPeriod;
    _nextPhase = _nextPhase === 'inhale' ? 'exhale' : 'inhale';
  }
  _schedulerTimer = setTimeout(_schedulerTick, LOOKAHEAD_MS);
}
```

**startPacer changes (line 51):**
- Keep `AppState.pacingFreq = pacingFreqHz;` (set before first tick — already implied by practice.js line 155 writing pacingFreq before calling startPacer)
- Remove local `const halfPeriod = 1 / (pacingFreqHz * 2);`
- Change `_schedulerTick(halfPeriod)` to `_schedulerTick()`

Before:
```javascript
export function startPacer(pacingFreqHz) {
  if (!_ctx) return;
  const halfPeriod = 1 / (pacingFreqHz * 2);
  _nextCueTime = _ctx.currentTime + 0.1;
  _nextPhase = 'inhale';
  AppState.pacerEpoch = _nextCueTime;
  _schedulerTick(halfPeriod);
}
```

After:
```javascript
export function startPacer(pacingFreqHz) {
  if (!_ctx) return;
  AppState.pacingFreq = pacingFreqHz;
  _nextCueTime = _ctx.currentTime + 0.1;
  _nextPhase = 'inhale';
  AppState.pacerEpoch = _nextCueTime;
  _schedulerTick();
}
```

**Critical:** The `halfPeriod` is sampled once per 25ms scheduler tick. All cues scheduled within that tick use the same halfPeriod. A frequency change mid-breath takes effect on the NEXT cue — no retroactive correction of already-enqueued cues. This is correct behavior.

**Anti-pattern to avoid:** Do NOT call `stopPacer()` + `startPacer()` to change frequency — that causes an audible gap and resets `pacerEpoch`, which breaks phase lock scoring. The dynamic read approach avoids this entirely.
